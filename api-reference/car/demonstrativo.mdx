---
title: "Demonstrativo Ambiental"
api: "GET /car/{codigo_car}/demonstrativo"
description: "Dados detalhados do demonstrativo ambiental do CAR (RL, APP, vegetacao nativa)"
---

## Descricao

Busca o demonstrativo ambiental completo do CAR no site oficial car.gov.br. Inclui dados de Reserva Legal, APP, vegetacao nativa, uso consolidado e sobreposicoes.

<ParamField path="codigo_car" type="string" required>
  Codigo do CAR.
</ParamField>

## Response

<ResponseField name="data" type="object">
  <Expandable title="data">
    <ResponseField name="demonstrativo" type="object">
      <Expandable title="demonstrativo">
        <ResponseField name="cabecalho" type="object">
          Dados gerais: codigo, municipio, estado, area_ha, modulos_fiscais, coordenadas, datas de registro/retificacao, status, condicao de analise.
        </ResponseField>

        <ResponseField name="areas" type="object">
          <Expandable title="areas">
            <ResponseField name="imovel" type="object">
              `area_liquida_ha`, `area_uso_consolidado_ha`
            </ResponseField>
            <ResponseField name="reserva_legal" type="object">
              `situacao`, `area_rl_averbada_ha`, `area_rl_proposta_ha`, `area_rl_em_app_ha`, `area_rl_minima_exigida_ha`, `area_rl_excedente_passivo_ha`, `area_rl_recompor_ha`, `area_rl_vetorizada_sobreposta_rvn_ha`
            </ResponseField>
            <ResponseField name="vegetacao_nativa" type="object">
              `area_rvn_ha` (remanescente), `area_aa_ha` (antropizada anterior a 2008), `area_ap_ha` (antropizada posterior)
            </ResponseField>
            <ResponseField name="app" type="object">
              `area_app_ha`, `area_app_em_aa_ha`, `area_app_em_ac_ha`, `area_app_recompor_ha`, `area_app_sobreposta_rvn_ha`
            </ResponseField>
            <ResponseField name="uso_restrito" type="object">
              `area_uso_restrito_ha`, `area_uso_restrito_recompor_ha`
            </ResponseField>
            <ResponseField name="sobreposicoes" type="object">
              `area_sobreposicao_outros_imoveis_ha`, `area_sobreposicao_terra_indigena_ha`, `area_sobreposicao_uc_ha`, `area_sobreposicao_assentamento_ha`
            </ResponseField>
          </Expandable>
        </ResponseField>

        <ResponseField name="restricoes" type="array">
          Lista de restricoes aplicaveis ao imovel.
        </ResponseField>

        <ResponseField name="tem_terra_indigena" type="boolean">
          Se o imovel tem sobreposicao com terra indigena.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
